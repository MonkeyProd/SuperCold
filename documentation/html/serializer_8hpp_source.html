16// This function serialize a key. It checks a string is a bare key and
17// escapes special characters if the string is not compatible to a bare key.
18// ```cpp
19// std::string k("non.bare.key"); // the key itself includes `.`s.
20// std::string formatted = toml::format_key(k);
21// assert(formatted == "\"non.bare.key\"");
22// ```
23//
24// This function is exposed to make it easy to write a user-defined serializer.
25// Since toml restricts characters available in a bare key, generally a string
26// should be escaped. But checking whether a string needs to be surrounded by
27// a `"` and escaping some special character is boring.
28template<typename charT, typename traits, typename Alloc>
29std::basic_string<charT, traits, Alloc>
30format_key(const std::basic_string<charT, traits, Alloc>& k)
31{
32 if(k.empty())
33 {
34 return std::string("\"\"");
35 }
36
37 // check the key can be a bare (unquoted) key
38 detail::location loc(k, std::vector<char>(k.begin(), k.end()));
39 detail::lex_unquoted_key::invoke(loc);
40 if(loc.iter() == loc.end())
41 {
42 return k; // all the tokens are consumed. the key is unquoted-key.
43 }
44
45 //if it includes special characters, then format it in a "quoted" key.
46 std::basic_string<charT, traits, Alloc> serialized("\"");
47 for(const char c : k)
48 {
49 switch(c)
50 {
51 case '\\': {serialized += "\\\\"; break;}
52 case '\"': {serialized += "\\\""; break;}
53 case '\b': {serialized += "\\b"; break;}
54 case '\t': {serialized += "\\t"; break;}
55 case '\f': {serialized += "\\f"; break;}
56 case '\n': {serialized += "\\n"; break;}
57 case '\r': {serialized += "\\r"; break;}
58 default : {serialized += c; break;}
59 }
60 }
61 serialized += "\"";
62 return serialized;
63}
64
65template<typename charT, typename traits, typename Alloc>
66std::basic_string<charT, traits, Alloc>
67format_keys(const std::vector<std::basic_string<charT, traits, Alloc>>& keys)
68{
69 if(keys.empty())
70 {
71 return std::string("\"\"");
72 }
73
74 std::basic_string<charT, traits, Alloc> serialized;
75 for(const auto& ky : keys)
76 {
77 serialized += format_key(ky);
78 serialized += charT('.');
79 }
80 serialized.pop_back(); // remove the last dot '.'
81 return serialized;
82}

105 serializer(const std::size_t w = 80u,
106 const int float_prec = std::numeric_limits<toml::floating>::max_digits10,
107 const bool can_be_inlined = false,
108 const bool no_comment = false,

110 const bool value_has_comment = false)
111 : can_be_inlined_(can_be_inlined), no_comment_(no_comment),

117 std::string operator()(const boolean_type& b) const
118 {
119 return b ? "true" : "false";
120 }
121 std::string operator()(const integer_type i) const
122 {
123 return std::to_string(i);
124 }
125 std::string operator()(const floating_type f) const
126 {
127 if(std::isnan(f))
128 {
129 if(std::signbit(f))
130 {
131 return std::string("-nan");
132 }
133 else
134 {
135 return std::string("nan");
136 }
137 }
138 else if(!std::isfinite(f))
139 {
140 if(std::signbit(f))
141 {
142 return std::string("-inf");
143 }
144 else
145 {
146 return std::string("inf");
147 }
148 }
149
150 const auto fmt = "%.*g";
151 const auto bsz = std::snprintf(nullptr, 0, fmt, this->float_prec_, f);
152 // +1 for null character(\0)
153 std::vector<char> buf(static_cast<std::size_t>(bsz + 1), '\0');
154 std::snprintf(buf.data(), buf.size(), fmt, this->float_prec_, f);
155
156 std::string token(buf.begin(), std::prev(buf.end()));
157 if(!token.empty() && token.back() == '.') // 1. => 1.0
158 {
159 token += '0';
160 }
161
162 const auto e = std::find_if(
163 token.cbegin(), token.cend(), [](const char c) noexcept -> bool {
164 return c == 'e' || c == 'E';
165 });
166 const auto has_exponent = (token.cend() != e);
167 const auto has_fraction = (token.cend() != std::find(
168 token.cbegin(), token.cend(), '.'));
169
170 if(!has_exponent && !has_fraction)
171 {
172 // the resulting value does not have any float specific part!
173 token += ".0";
174 }
175 return token;
176 }
177 std::string operator()(const string_type& s) const
178 {
179 if(s.kind == string_t::basic)
180 {
181 if((std::find(s.str.cbegin(), s.str.cend(), '\n') != s.str.cend() ||
182 std::find(s.str.cbegin(), s.str.cend(), '\"') != s.str.cend()) &&
183 this->width_ != (std::numeric_limits<std::size_t>::max)())
184 {
185 // if linefeed or double-quote is contained,
186 // make it multiline basic string.
187 const auto escaped = this->escape_ml_basic_string(s.str);
188 std::string open("\"\"\"");
189 std::string close("\"\"\"");
190 if(escaped.find('\n') != std::string::npos ||
191 this->width_ < escaped.size() + 6)
192 {
193 // if the string body contains newline or is enough long,
194 // add newlines after and before delimiters.
195 open += "\n";
196 close = std::string("\\\n") + close;
197 }
198 return open + escaped + close;
199 }
200
201 // no linefeed. try to make it oneline-string.
202 std::string oneline = this->escape_basic_string(s.str);
203 if(oneline.size() + 2 < width_ || width_ < 2)
204 {
205 const std::string quote("\"");
206 return quote + oneline + quote;
207 }
208
209 // the line is too long compared to the specified width.
210 // split it into multiple lines.
211 std::string token("\"\"\"\n");
212 while(!oneline.empty())
213 {
214 if(oneline.size() < width_)
215 {
216 token += oneline;
217 oneline.clear();
218 }
219 else if(oneline.at(width_-2) == '\\')
220 {
221 token += oneline.substr(0, width_-2);
222 token += "\\\n";
223 oneline.erase(0, width_-2);
224 }
225 else
226 {
227 token += oneline.substr(0, width_-1);
228 token += "\\\n";
229 oneline.erase(0, width_-1);
230 }
231 }
232 return token + std::string("\\\n\"\"\"");
233 }
234 else // the string `s` is literal-string.
235 {
236 if(std::find(s.str.cbegin(), s.str.cend(), '\n') != s.str.cend() ||
237 std::find(s.str.cbegin(), s.str.cend(), '\'') != s.str.cend() )
238 {
239 std::string open("'''");
240 if(this->width_ + 6 < s.str.size())
241 {
242 open += '\n'; // the first newline is ignored by TOML spec
243 }
244 const std::string close("'''");
245 return open + s.str + close;
246 }
247 else
248 {
249 const std::string quote("'");
250 return quote + s.str + quote;
251 }
252 }
253 }

280 std::string operator()(const array_type& v) const
281 {
282 if(v.empty())
283 {
284 return std::string("[]");
285 }
286 if(this->is_array_of_tables(v))
287 {
288 return make_array_of_tables(v);
289 }
290
291 // not an array of tables. normal array.
292 // first, try to make it inline if none of the elements have a comment.
293 if( ! this->has_comment_inside(v))
294 {
295 const auto inl = this->make_inline_array(v);
296 if(inl.size() < this->width_ &&
297 std::find(inl.cbegin(), inl.cend(), '\n') == inl.cend())
298 {
299 return inl;
300 }
301 }
302
303 // if the length exceeds this->width_, print multiline array.
304 // key = [
305 // # ...
306 // 42,
307 // ...
308 // ]
309 std::string token;
310 std::string current_line;
311 token += "[\n";
312 for(const auto& item : v)
313 {
314 if( ! item.comments().empty() && !no_comment_)
315 {
316 // if comment exists, the element must be the only element in the line.
317 // e.g. the following is not allowed.
318 // ```toml
319 // array = [
320 // # comment for what?
321 // 1, 2, 3, 4, 5
322 // ]
323 // ```
324 if(!current_line.empty())
325 {
326 if(current_line.back() != '\n')
327 {
328 current_line += '\n';
329 }
330 token += current_line;
331 current_line.clear();
332 }
333 for(const auto& c : item.comments())
334 {
335 token += '#';
336 token += c;
337 token += '\n';
338 }
339 token += toml::visit(*this, item);
340 if(!token.empty() && token.back() == '\n') {token.pop_back();}
341 token += ",\n";
342 continue;
343 }
344 std::string next_elem;
345 if(item.is_table())
346 {
347 serializer ser(*this);
348 ser.can_be_inlined_ = true;
349 ser.width_ = (std::numeric_limits<std::size_t>::max)();
350 next_elem += toml::visit(ser, item);
351 }
352 else
353 {
354 next_elem += toml::visit(*this, item);
355 }
356
357 // comma before newline.
358 if(!next_elem.empty() && next_elem.back() == '\n') {next_elem.pop_back();}
359
360 // if current line does not exceeds the width limit, continue.
361 if(current_line.size() + next_elem.size() + 1 < this->width_)
362 {
363 current_line += next_elem;
364 current_line += ',';
365 }
366 else if(current_line.empty())
367 {
368 // if current line was empty, force put the next_elem because
369 // next_elem is not splittable
370 token += next_elem;
371 token += ",\n";
372 // current_line is kept empty
373 }
374 else // reset current_line
375 {
376 assert(current_line.back() == ',');
377 token += current_line;
378 token += '\n';
379 current_line = next_elem;
380 current_line += ',';
381 }
382 }
383 if(!current_line.empty())
384 {
385 if(!current_line.empty() && current_line.back() != '\n')
386 {
387 current_line += '\n';
388 }
389 token += current_line;
390 }
391 token += "]\n";
392 return token;
393 }

395 // templatize for any table-like container
396 std::string operator()(const table_type& v) const
397 {
398 // if an element has a comment, then it can't be inlined.
399 // table = {# how can we write a comment for this? key = "value"}
400 if(this->can_be_inlined_ && !(this->has_comment_inside(v)))
401 {
402 std::string token;
403 if(!this->keys_.empty())
404 {
405 token += format_key(this->keys_.back());
406 token += " = ";
407 }
408 token += this->make_inline_table(v);
409 if(token.size() < this->width_ &&
410 token.end() == std::find(token.begin(), token.end(), '\n'))
411 {
412 return token;
413 }
414 }
415
416 std::string token;
417 if(!keys_.empty())
418 {
419 token += '[';
420 token += format_keys(keys_);
421 token += "]\n";
422 }
423 token += this->make_multiline_table(v);
424 return token;
425 }

429 std::string escape_basic_string(const std::string& s) const
430 {
431 //XXX assuming `s` is a valid utf-8 sequence.
432 std::string retval;
433 for(const char c : s)
434 {
435 switch(c)
436 {
437 case '\\': {retval += "\\\\"; break;}
438 case '\"': {retval += "\\\""; break;}
439 case '\b': {retval += "\\b"; break;}
440 case '\t': {retval += "\\t"; break;}
441 case '\f': {retval += "\\f"; break;}
442 case '\n': {retval += "\\n"; break;}
443 case '\r': {retval += "\\r"; break;}
444 default :
445 {
446 if((0x00 <= c && c <= 0x08) || (0x0A <= c && c <= 0x1F) || c == 0x7F)
447 {
448 retval += "\\u00";
449 retval += char(48 + (c / 16));
450 retval += char((c % 16 < 10 ? 48 : 55) + (c % 16));
451 }
452 else
453 {
454 retval += c;
455 }
456 }
457 }
458 }
459 return retval;
460 }
461
462 std::string escape_ml_basic_string(const std::string& s) const
463 {
464 std::string retval;
465 for(auto i=s.cbegin(), e=s.cend(); i!=e; ++i)
466 {
467 switch(*i)
468 {
469 case '\\': {retval += "\\\\"; break;}
470 // One or two consecutive "s are allowed.
471 // Later we will check there are no three consecutive "s.
472 // case '\"': {retval += "\\\""; break;}
473 case '\b': {retval += "\\b"; break;}
474 case '\t': {retval += "\\t"; break;}
475 case '\f': {retval += "\\f"; break;}
476 case '\n': {retval += "\n"; break;}
477 case '\r':
478 {
479 if(std::next(i) != e && *std::next(i) == '\n')
480 {
481 retval += "\r\n";
482 ++i;
483 }
484 else
485 {
486 retval += "\\r";
487 }
488 break;
489 }
490 default :
491 {
492 const auto c = *i;
493 if((0x00 <= c && c <= 0x08) || (0x0A <= c && c <= 0x1F) || c == 0x7F)
494 {
495 retval += "\\u00";
496 retval += char(48 + (c / 16));
497 retval += char((c % 16 < 10 ? 48 : 55) + (c % 16));
498 }
499 else
500 {
501 retval += c;
502 }
503 }
504
505 }
506 }
507 // Only 1 or 2 consecutive `"`s are allowed in multiline basic string.
508 // 3 consecutive `"`s are considered as a closing delimiter.
509 // We need to check if there are 3 or more consecutive `"`s and insert
510 // backslash to break them down into several short `"`s like the `str6`
511 // in the following example.
512 // ```toml
513 // str4 = """Here are two quotation marks: "". Simple enough."""
514 // # str5 = """Here are three quotation marks: """.""" # INVALID
515 // str5 = """Here are three quotation marks: ""\"."""
516 // str6 = """Here are fifteen quotation marks: ""\"""\"""\"""\"""\"."""
517 // ```
518 auto found_3_quotes = retval.find("\"\"\"");
519 while(found_3_quotes != std::string::npos)
520 {
521 retval.replace(found_3_quotes, 3, "\"\"\\\"");
522 found_3_quotes = retval.find("\"\"\"");
523 }
524 return retval;
525 }

527 // if an element of a table or an array has a comment, it cannot be inlined.
528 bool has_comment_inside(const array_type& a) const noexcept
529 {
530 // if no_comment is set, comments would not be written.
531 if(this->no_comment_) {return false;}
532
533 for(const auto& v : a)
534 {
535 if(!v.comments().empty()) {return true;}
536 }
537 return false;
538 }
539 bool has_comment_inside(const table_type& t) const noexcept
540 {
541 // if no_comment is set, comments would not be written.
542 if(this->no_comment_) {return false;}
543
544 for(const auto& kv : t)
545 {
546 if(!kv.second.comments().empty()) {return true;}
547 }
548 return false;
549 }

551 std::string make_inline_array(const array_type& v) const
552 {
553 assert(!has_comment_inside(v));
554 std::string token;
555 token += '[';
556 bool is_first = true;
557 for(const auto& item : v)
558 {
559 if(is_first) {is_first = false;} else {token += ',';}
560 token += visit(serializer(
561 (std::numeric_limits<std::size_t>::max)(), this->float_prec_,
562 /* inlined */ true, /*no comment*/ false, /*keys*/ {},
563 /*has_comment*/ !item.comments().empty()), item);
564 }
565 token += ']';
566 return token;
567 }
568
569 std::string make_inline_table(const table_type& v) const
570 {
571 assert(!has_comment_inside(v));
572 assert(this->can_be_inlined_);
573 std::string token;
574 token += '{';
575 bool is_first = true;
576 for(const auto& kv : v)
577 {
578 // in inline tables, trailing comma is not allowed (toml-lang #569).
579 if(is_first) {is_first = false;} else {token += ',';}
580 token += format_key(kv.first);
581 token += '=';
582 token += visit(serializer(
583 (std::numeric_limits<std::size_t>::max)(), this->float_prec_,
584 /* inlined */ true, /*no comment*/ false, /*keys*/ {},
585 /*has_comment*/ !kv.second.comments().empty()), kv.second);
586 }
587 token += '}';
588 return token;
589 }
590
591 std::string make_multiline_table(const table_type& v) const
592 {
593 std::string token;
594
595 // print non-table elements first.
596 // ```toml
597 // [foo] # a table we're writing now here
598 // key = "value" # <- non-table element, "key"
599 // # ...
600 // [foo.bar] # <- table element, "bar"
601 // ```
602 // because after printing [foo.bar], the remaining non-table values will
603 // be assigned into [foo.bar], not [foo]. Those values should be printed
604 // earlier.
605 for(const auto& kv : v)
606 {
607 if(kv.second.is_table() || is_array_of_tables(kv.second))
608 {
609 continue;
610 }
611
612 token += write_comments(kv.second);
613
614 const auto key_and_sep = format_key(kv.first) + " = ";
615 const auto residual_width = (this->width_ > key_and_sep.size()) ?
616 this->width_ - key_and_sep.size() : 0;
617 token += key_and_sep;
618 token += visit(serializer(residual_width, this->float_prec_,
619 /*can be inlined*/ true, /*no comment*/ false, /*keys*/ {},
620 /*has_comment*/ !kv.second.comments().empty()), kv.second);
621
622 if(token.back() != '\n')
623 {
624 token += '\n';
625 }
626 }
627
628 // normal tables / array of tables
629
630 // after multiline table appeared, the other tables cannot be inline
631 // because the table would be assigned into the table.
632 // [foo]
633 // ...
634 // bar = {...} # <- bar will be a member of [foo].
635 bool multiline_table_printed = false;
636 for(const auto& kv : v)
637 {
638 if(!kv.second.is_table() && !is_array_of_tables(kv.second))
639 {
640 continue; // other stuff are already serialized. skip them.
641 }
642
643 std::vector<toml::key> ks(this->keys_);
644 ks.push_back(kv.first);
645
646 auto tmp = visit(serializer(this->width_, this->float_prec_,
647 !multiline_table_printed, this->no_comment_, ks,
648 /*has_comment*/ !kv.second.comments().empty()), kv.second);
649
650 // If it is the first time to print a multi-line table, it would be
651 // helpful to separate normal key-value pair and subtables by a
652 // newline.
653 // (this checks if the current key-value pair contains newlines.
654 // but it is not perfect because multi-line string can also contain
655 // a newline. in such a case, an empty line will be written) TODO
656 if((!multiline_table_printed) &&
657 std::find(tmp.cbegin(), tmp.cend(), '\n') != tmp.cend())
658 {
659 multiline_table_printed = true;
660 token += '\n'; // separate key-value pairs and subtables
661
662 token += write_comments(kv.second);
663 token += tmp;
664
665 // care about recursive tables (all tables in each level prints
666 // newline and there will be a full of newlines)
667 if(tmp.substr(tmp.size() - 2, 2) != "\n\n" &&
668 tmp.substr(tmp.size() - 4, 4) != "\r\n\r\n" )
669 {
670 token += '\n';
671 }
672 }
673 else
674 {
675 token += write_comments(kv.second);
676 token += tmp;
677 token += '\n';
678 }
679 }
680 return token;
681 }
682
683 std::string make_array_of_tables(const array_type& v) const
684 {
685 // if it's not inlined, we need to add `[[table.key]]`.
686 // but if it can be inlined, we can format it as the following.
687 // ```
688 // table.key = [
689 // {...},
690 // # comment
691 // {...},
692 // ]
693 // ```
694 // This function checks if inlinization is possible or not, and then
695 // format the array-of-tables in a proper way.
696 //
697 // Note about comments:
698 //
699 // If the array itself has a comment (value_has_comment_ == true), we
700 // should try to make it inline.
701 // ```toml
702 // # comment about array
703 // array = [
704 // # comment about table element
705 // {of = "table"}
706 // ]
707 // ```
708 // If it is formatted as a multiline table, the two comments becomes
709 // indistinguishable.
710 // ```toml
711 // # comment about array
712 // # comment about table element
713 // [[array]]
714 // of = "table"
715 // ```
716 // So we need to try to make it inline, and it force-inlines regardless
717 // of the line width limit.
718 // It may fail if the element of a table has comment. In that case,
719 // the array-of-tables will be formatted as a multiline table.
720 if(this->can_be_inlined_ || this->value_has_comment_)
721 {
722 std::string token;
723 if(!keys_.empty())
724 {
725 token += format_key(keys_.back());
726 token += " = ";
727 }
728
729 bool failed = false;
730 token += "[\n";
731 for(const auto& item : v)
732 {
733 // if an element of the table has a comment, the table
734 // cannot be inlined.
735 if(this->has_comment_inside(item.as_table()))
736 {
737 failed = true;
738 break;
739 }
740 // write comments for the table itself
741 token += write_comments(item);
742
743 const auto t = this->make_inline_table(item.as_table());
744
745 if(t.size() + 1 > width_ || // +1 for the last comma {...},
746 std::find(t.cbegin(), t.cend(), '\n') != t.cend())
747 {
748 // if the value itself has a comment, ignore the line width limit
749 if( ! this->value_has_comment_)
750 {
751 failed = true;
752 break;
753 }
754 }
755 token += t;
756 token += ",\n";
757 }
758
759 if( ! failed)
760 {
761 token += "]\n";
762 return token;
763 }
764 // if failed, serialize them as [[array.of.tables]].
765 }
766
767 std::string token;
768 for(const auto& item : v)
769 {
770 token += write_comments(item);
771 token += "[[";
772 token += format_keys(keys_);
773 token += "]]\n";

871template<typename charT, typename traits, typename C,
872 template<typename ...> class M, template<typename ...> class V>
873std::basic_ostream<charT, traits>&
874operator<<(std::basic_ostream<charT, traits>& os, const basic_value<C, M, V>& v)
875{
876 using value_type = basic_value<C, M, V>;
877
878 // get status of std::setw().
879 const auto w = static_cast<std::size_t>(os.width());
880 const int fprec = static_cast<int>(os.precision());
881 os.width(0);
882
883 // by default, iword is initialized by 0. And by default, toml11 outputs
884 // comments. So `0` means showcomment. 1 means nocommnet.
885 const bool no_comment = (1 == os.iword(detail::comment_index(os)));
886
887 if(!no_comment && v.is_table() && !v.comments().empty())
888 {
889 os << v.comments();
890 os << '\n'; // to split the file comment from the first element
891 }
892 // the root object can't be an inline table. so pass `false`.
893 const auto serialized = visit(serializer<value_type>(w, fprec, no_comment, false), v);
894 os << serialized;
895
896 // if v is a non-table value, and has only one comment, then
897 // put a comment just after a value. in the following way.
898 //
899 // ```toml
900 // key = "value" # comment.
901 // ```
902 //
903 // Since the top-level toml object is a table, one who want to put a
904 // non-table toml value must use this in a following way.
905 //
906 // ```cpp
907 // toml::value v;
908 // std::cout << "user-defined-key = " << v << std::endl;
909 // ```
910 //
911 // In this case, it is impossible to put comments before key-value pair.
912 // The only way to preserve comments is to put all of them after a value.
913 if(!no_comment && !v.is_table() && !v.comments().empty())
914 {
915 os << " #";
916 for(const auto& c : v.comments()) {os << c;}
917 }
918 return os;
919}
